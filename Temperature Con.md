//  Part 1: Temperature Converter  - Fahrenheit to Celsius
#include <iostream>
using namespace std;

int main() {
    cout << "Enter the Fahrenheit: ";
    int Fahrenheit;
    cin >> Fahrenheit;
    double Celsius = (Fahrenheit - 32)/1.8;
    cout << "Celsius: " << Celsius;
    return 0;
}
